정보는 아래의 db에 저장이 되어야 해



CREATE TABLE `wb_att_type` (
	`att_type_idx` INT(11) NOT NULL AUTO_INCREMENT,
	`att_type_name` VARCHAR(20) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	`att_type_category_name` VARCHAR(20) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	`att_type_category_idx` INT(11) NULL DEFAULT NULL,
	`att_type_nickname` VARCHAR(20) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	`group_id` INT(11) NULL DEFAULT NULL,
	PRIMARY KEY (`att_type_idx`) USING BTREE
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=87
;
